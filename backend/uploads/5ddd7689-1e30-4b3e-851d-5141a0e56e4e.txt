#!/usr/bin/env bash 

set -euo pipefail



# MoME end-to-end runner

# Usage examples:

#   bash run_mome_pipeline.sh prepare-data

#   bash run_mome_pipeline.sh stage1

#   bash run_mome_pipeline.sh stage2-baseline

#   bash run_mome_pipeline.sh stage2-soft

#   bash run_mome_pipeline.sh stage2-seq

#   bash run_mome_pipeline.sh stage2-parallel-tmux

#   bash run_mome_pipeline.sh full-seq



ROOT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"

cd "$ROOT_DIR"



# Adjust this if your conda env name changes.

CONDA_ENV="mome"



# Config paths

CFG_STAGE1="projects/configs/mome/stage1_trainval.py"

CFG_STAGE2_BASE="projects/configs/mome/stage2_baseline_trainval.py"

CFG_STAGE2_SOFT="projects/configs/mome/stage2_soft_mome_trainval.py"



# Data paths

NUSC_ROOT="data/nuscenes"

INFO_TRAIN="$NUSC_ROOT/nuscenes_infos_train.pkl"

DBINFO_TRAIN="$NUSC_ROOT/nuscenes_dbinfos_train.pkl"



activate_env() {

  # shellcheck disable=SC1091

  eval "$(conda shell.bash hook 2>/dev/null)"

  conda activate "$CONDA_ENV"

  export PYTHONPATH="$ROOT_DIR:$ROOT_DIR/tools:${PYTHONPATH:-}"

}



check_gpu() {

  if command -v nvidia-smi >/dev/null 2>&1; then

    echo "[GPU] Current GPU usage:"

    nvidia-smi --query-gpu=index,name,memory.used,memory.total,utilization.gpu --format=csv,noheader || true

  fi

}



prepare_data() {

  activate_env

  echo "[STEP] Running dataset info generation via tools/create_data.sh"

  bash tools/create_data.sh



  if [[ ! -f "$DBINFO_TRAIN" ]]; then

    echo "[STEP] GT DB missing. Rebuilding nuscenes_dbinfos_train.pkl"

    python -c "

import importlib

importlib.import_module('projects.mmdet3d_plugin')

from data_converter.create_gt_database import create_groundtruth_database

create_groundtruth_database('CustomNuScenesDataset', './data/nuscenes/', 'nuscenes', './data/nuscenes/nuscenes_infos_train.pkl')

"

  else

    echo "[OK] Found GT DB: $DBINFO_TRAIN"

  fi



  echo "[DONE] Data prep complete"

}



train_stage1() {

  activate_env

  check_gpu

  echo "[STEP] Stage 1 training (single GPU)"

  echo "[CMD] bash tools/dist_train.sh $CFG_STAGE1 1"

  bash tools/dist_train.sh "$CFG_STAGE1" 1

}



train_stage2_baseline() {

  activate_env

  check_gpu

  echo "[STEP] Stage 2 baseline training (2 GPUs)"

  echo "[CMD] bash tools/dist_train.sh $CFG_STAGE2_BASE 2"

  bash tools/dist_train.sh "$CFG_STAGE2_BASE" 2

}



train_stage2_soft() {

  activate_env

  check_gpu

  echo "[STEP] Stage 2 Soft-MoME novelty training (2 GPUs)"

  echo "[CMD] bash tools/dist_train.sh $CFG_STAGE2_SOFT 2"

  bash tools/dist_train.sh "$CFG_STAGE2_SOFT" 2

}



train_stage2_seq() {

  train_stage2_baseline

  train_stage2_soft

}



train_stage2_parallel_tmux() {

  activate_env

  check_gpu



  if ! command -v tmux >/dev/null 2>&1; then

    echo "[ERROR] tmux not found. Install tmux or use stage2-seq mode."

    exit 1

  fi



  echo "[STEP] Launching Stage 2 baseline and novelty in parallel tmux sessions"

  tmux new-session -d -s mome_stage2_base \

    "bash -lc 'cd \"$ROOT_DIR\" && eval \"\

\$(conda shell.bash hook 2>/dev/null)\" && conda activate \"$CONDA_ENV\" && export PYTHONPATH=\"$ROOT_DIR:$ROOT_DIR/tools:\$PYTHONPATH\" && bash tools/dist_train.sh \"$CFG_STAGE2_BASE\" 1 2>&1 | tee stage2_baseline_train.log'"



  tmux new-session -d -s mome_stage2_soft \

    "bash -lc 'cd \"$ROOT_DIR\" && eval \"\

\$(conda shell.bash hook 2>/dev/null)\" && conda activate \"$CONDA_ENV\" && export PYTHONPATH=\"$ROOT_DIR:$ROOT_DIR/tools:\$PYTHONPATH\" && CUDA_VISIBLE_DEVICES=1 bash tools/dist_train.sh \"$CFG_STAGE2_SOFT\" 1 2>&1 | tee stage2_soft_train.log'"



  echo "[DONE] Started tmux sessions: mome_stage2_base, mome_stage2_soft"

  echo "[TIP] Monitor with: tmux attach -t mome_stage2_base"

  echo "[TIP] Monitor with: tmux attach -t mome_stage2_soft"

}



full_seq() {

  prepare_data

  train_stage1

  train_stage2_seq

}



usage() {

  cat <<EOF

Usage: bash run_mome_pipeline.sh <mode>



Modes:

  prepare-data          Run create_data + ensure GT DB exists

  stage1                Run Stage 1 training (single GPU)

  stage2-baseline       Run Stage 2 baseline training

  stage2-soft           Run Stage 2 novelty (Soft-MoME) training

  stage2-seq            Run baseline then novelty sequentially

  stage2-parallel-tmux  Run baseline and novelty in parallel tmux sessions

  full-seq              Run full pipeline sequentially

EOF

}



MODE="${1:-}"

case "$MODE" in

  prepare-data) prepare_data ;;

  stage1) train_stage1 ;;

  stage2-baseline) train_stage2_baseline ;;

  stage2-soft) train_stage2_soft ;;

  stage2-seq) train_stage2_seq ;;

  stage2-parallel-tmux) train_stage2_parallel_tmux ;;

  full-seq) full_seq ;;

  *) usage; exit 1 ;;

esac
